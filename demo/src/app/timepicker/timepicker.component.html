Single datepicker with custom invalid date & custom class on date

Html code:


<mat-form-field>
    <input
        type="text"
        matInput
        ngxDaterangepickerMd
        [timePicker]="true"
        [timePickerSeconds]="true"
        [timePickerIncrement]="15"
        [timePicker24Hour]="true"
        [(ngModel)]="selected"
        placeholder="Choose date"
        class="form-control"
    />
</mat-form-field>

Typescript code:

export class SingleDatepickerComponent {{ '{' }}
        selected: any;
        
        isInvalidDate(date) {{ '{' }}
          return date.weekday() === 0;
        {{'}'}}
