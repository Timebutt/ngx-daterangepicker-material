Simple daterangepicker

Example:


<mat-form-field>
    <input
        type="text"
        name="daterangepicker"
        ngxDaterangepickerMd
        matInput
        [(ngModel)]="selected"
        class="form-control"
        placeholder="Choose date"
        [showDropdowns]="true"
        [lockStartDate]="false"
        [customRangeDirection]="false"
        (ngModelChange)="ngModelChange($event)"
        (change)="change($event)"
        showCancel="true"
    />
</mat-form-field>
